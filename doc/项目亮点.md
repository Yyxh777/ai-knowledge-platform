# 1. 知识库搭建

- 通过业务服务编排模式搭建知识库。业务服务负责文件生命周期管理，AI服务负责向量化和检索，通过外部ID关联
- milvus数据库通过 外部ID_分块顺序 作为数据主键，且为常用查询字段建立标量索引，在便于调试的同时又让查询不走全表扫描



## 1.1. Milvus设计

### 1.1.1. 创建Milvus集合（字段设计、索引设计）

```python
milvus_client = MilvusClient(uri=MILVUS_URI)
	# 1) 创建Shema参数
    # auto_id=False 主键id不自增
    # enable_dynamic_field True时可以插入没有定义的字段但不建索引，False则只能插入定义的字段
    schema = milvus_client.create_schema(auto_id=False, enable_dynamic_field=False)
    
    schema.add_field("id",DataType.VARCHAR, is_primary=True, max_length=128)  
    # 2) 向量字段参数
    # 向量在milvus中是一个浮点数数组，而FLOAT_VECTOR=32位浮点数数组
    # 还有BINARY_VECTOR=二进制向量，精度低体积小
    # dim 向量维度，即用1024个数字来标识一个token/词，值是根据向量模型而变化，如千问则是1024，
    schema.add_field("vector",       DataType.FLOAT_VECTOR, dim=1024)
    # max_length=65535 是存储长度的最大值，若更大，则换DataType.JSON存储
    # 或者Milvus存储向量及其摘要，内容存MySQL/对象存储中
    schema.add_field("text",         DataType.VARCHAR, max_length=65535)
    schema.add_field("record_id",    DataType.VARCHAR, max_length=64)   # 用于按文件删除
    schema.add_field("type",         DataType.VARCHAR, max_length=32)   # policy / tech
    schema.add_field("access_level", DataType.VARCHAR, max_length=32)   # public / internal / hr_only
    
    # 给 type 和 access_level 建标量索引，让 filter 查询不走全表扫描
    index_params = milvus_client.prepare_index_params()
    # 3）向量相似度计算方式有IP、LP
    # IP 是内积，也就是余弦相似度，俩向量相乘，越大越相似
    # LP 是欧式距离，俩向量几何距离 越小越相似
    # 由于向量模型已经对向量进行归一化(长度为1)，在此基础上IP语义检索比LP好
    # 4）向量索引类型：代码下补充
    index_params.add_index("vector",       metric_type="IP", index_type="IVF_FLAT", params={"nlist": 128})
    # 5）标量索引类型：代码下补充
    index_params.add_index("record_id",    index_type="INVERTED")  # 删除时用
    index_params.add_index("type",         index_type="BITMAP")  # filter 过滤时用
    index_params.add_index("access_level", index_type="BITMAP")  # 权限 filter 时用

    milvus_client.create_collection(
        collection_name=collection_name,
        schema=schema,
        index_params=index_params,
    )

    return True
```



> 向量字段索引类型

| index_type | 原理                   | 速度 | 精度       | 适合场景                 |
| :--------- | :--------------------- | :--- | :--------- | :----------------------- |
| FLAT       | 暴力全量遍历           | 极慢 | 100%       | 数据量 < 1万             |
| IVF_FLAT   | 先分桶，再在桶内暴力搜 | 快   | 高（99%+） | 数据量 1万~100万         |
| IVF_SQ8    | 分桶+量化压缩          | 更快 | 略低       | 数据量 > 100万，内存敏感 |
| HNSW       | 图结构近邻搜索         | 最快 | 高         | 高并发、低延迟           |

> 标量字段索引类型

| index_type | 适用场景                               |
| :--------- | :------------------------------------- |
| INVERTED   | VARCHAR/字符串精确匹配                 |
| BITMAP     | 低基数字段（如性别、状态，只有几个值） |
| STL_SORT   | 数值范围查询（如时间戳 > 某个值）      |

# 2. WebSocket 设计

- 通过WebSocket建立客户端与服务端的连接，客户端在第一次发送请求发送token进行校验，同时实现流式输出



> 概念

普通HTTP请求是“一问一答”型，客户端发起请求->服务端响应->连接关闭

WebSocket则是“持久连接”型，客户端和服务端建立一次连接之后，可以双向随时发送消息，连接保持到某一方主动关闭连接

```
HTTP:  客户端 →→→ 服务器 (请求)
             ←←← 服务器 (响应)
             连接关闭

WebSocket: 客户端 ←→ 服务器 (建立连接)
                  →→→ (客户端发消息)
                  ←←← (服务器发消息)
                  →→→ (客户端发消息)
                  ←←← (服务器一直流式返回 token)
                  连接保持，直到任意一方关闭
```

```py
async def safe_send(websocket: WebSocket, data: dict) -> bool:
    """
    安全发送 WebSocket 消息。
    连接已断开时静默忽略，不抛出异常。
    返回 True=发送成功，False=连接已断开
    """
    try:
        await websocket.send_json(data)
        return True
    except Exception:
        return False


@router.websocket("/ws/chat")
async def websocket_chat(websocket: WebSocket):
    # 接受客户端连接
    await websocket.accept()

    # 值初始化，防止连接断开时 except 出现异常，访问到未赋值变量
    user_info: Dict = {}
    token: str = ""

    try:
        # 获取客户端第一条消息（token认证消息）
        auth_data = await websocket.receive_text()
        auth_message = json.loads(auth_data)           
        token = auth_message.get("token", "")          

        # 若token为空，则返回
        if not token:
            await safe_send(websocket, {"type": "error", "content": "缺少 token，请先登录"})
            await websocket.close()
            return

        # token发送至Java服务进行校验
        is_valid, user_info = await validate_token(token)   
        if not is_valid:                                     
            await safe_send(websocket, {"type": "error", "content": "token 无效或已过期"})
            await websocket.close()
            return

        # 验证成功，发送确认消息
        await safe_send(websocket, {
            "type": "auth_success",
            "user": {
                "user_id": user_info["user_id"],
                "account": user_info["account"],
            }
        })

        print(f"WebSocket 用户 {user_info['account']}({user_info['user_id']}) 已连接")

        # 成功则进行消息循环
        while True:
            # 等待客户端消息
            data = await websocket.receive_text()
            message_data = json.loads(data)             
            # 提取问题内容
            query = message_data.get("message", "")
            if not query:
                continue

            await safe_send(websocket, {"type": "start"})

            # 发送给 Agent
            try:
                async for chunk in chat_with_rag_stream(    
                    query,
                    user_info=user_info,
                    access_token=token
                ):
                    # 流式响应持续获取内容并发送给客户端
                    await safe_send(websocket, {"type": "token", "content": chunk})  

                # 生成结束，放在 try 内，只有成功才发 end
                await safe_send(websocket, {"type": "end"})  

            except Exception as e:
                print(f"处理消息错误: {str(e)}")
                await safe_send(websocket, {
                    "type": "error",
                    "content": f"处理消息时出错: {str(e)}"
                })

    except WebSocketDisconnect:                                          
        print(f"用户 {user_info.get('account', '未认证用户')} 断开连接") 
    except json.JSONDecodeError:
        await safe_send(websocket, {"type": "error", "content": "消息格式错误"})
        await websocket.close()
    except Exception as e:
        print(f"WebSocket 错误: {str(e)}")
        await websocket.close()
```
